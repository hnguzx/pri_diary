#import <Foundation/Foundation>;

@interface PdDiary : NSObject

//日记唯一标识
@property (nonatomic, strong) NSInteger diaryId;
//日记所属用户id
@property (nonatomic, strong) NSInteger userId;
//日记封面
@property (nonatomic, assign) NSString diaryPhoto;
//日记标题
@property (nonatomic, assign) NSString diaryTitle;
//当天天气 [SUNNY_DAY,SUNNY_NIGHT,HAIL,HEAVY_SNOW,HEAVY_RAIN,HEAVY_RAIN_DAY,HEAVY_RAIN_NIGHT,FLOATING_DUST,THUNDER_AND_LIGHTNING,HAZE,TYPHOON,FOG,LIGHT_SNOW,LIGHT_RAIN,CLOUDY_DAY,SNOW_SHOWER_DAY,SNOW_SHOWER_NIGHT,SHOWER_DAY,SHOWER_NIGHT,RAISE_SAND]
@property (nonatomic, strong) enum diaryWeather;
//当天心情 [HAPPY,ENRICH,SURPRISED,PROUD,WARM,TOUCH,SORRY,AGITATED,LOST,LONELY,ANGRY,EMBARRASSED,INJUSTICE,SWEET,DREAM,TIRED,ESCAPE,IT_IS_A_LONG_STORY]
@property (nonatomic, strong) enum diaryMood;
//当天主要事件 [MOVIE,GAME,FAMILY,TOURISM,RELATIONSHIP,FOOD,MUSIC,WORK,REST,LOVE,SHOPPING,ADVENTURE,IMAGINATION,LEARNING,READING,MOVEMENT,ADJUST,FITNESS,SICK,STAR,PET,AFTER_PLAY,CARTOON,INTERNET,ONLINE_SHOPPING,EAT_THE_MELON,STAY_UP_LATE,DRAWING,COOKING,PLANT,BALANCE,BECOME_BEAUTIFUL,MILK_TEA,KEEPING_IN_GOOD_HEALTH,CLOTHES,OTHER]
@property (nonatomic, strong) enum diaryEvent;
//日记详情
@property (nonatomic, assign) NSString diaryContent;
//当天日记所在位置
@property (nonatomic, assign) NSString diaryLocation;
//日记记录经度
@property (nonatomic, assign) NSString diaryLongitude;
//日记记录纬度
@property (nonatomic, assign) NSString diaryLatitude;
//日记创建时间
@property (nonatomic, assign) NSString diaryCreateTime;
//日记创建日期
@property (nonatomic, assign) NSString diaryCreateDay;
//日记最后更新时间
@property (nonatomic, assign) NSString diaryUpdateTime;


@end
