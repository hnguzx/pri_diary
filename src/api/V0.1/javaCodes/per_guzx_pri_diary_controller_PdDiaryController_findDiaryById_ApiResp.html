public class ApiResp implements Serializable{

    private int code; //错误码
    private String msg; //错误信息
    private PdDiary data; //响应内容，默认为null

    public int getCode(){
        return code;
    }

    public void setCode(int code){
        this.code = code;
    }

    public String getMsg(){
        return msg;
    }

    public void setMsg(String msg){
        this.msg = msg;
    }

    public PdDiary getData(){
        return data;
    }

    public void setData(PdDiary data){
        this.data = data;
    }

}
