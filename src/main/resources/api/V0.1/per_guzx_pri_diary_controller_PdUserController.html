用户注册

请求URL

/user/insertUser/{verifyCode} POST

请求体 application/json

{
	"userId":"int //用户唯一性标识",
	"userName":"string //用户昵称【必须】",
	"userPassword":"string //登录密码【必须】",
	"userBirthday":"string //用户生日【必须】",
	"userSex":"enum //用户性别 [FEMALE,MALE]【必须】",
	"userState":"enum //用户状态 [ACTIVATION,CANCELLATION,INACTIVATION]",
	"userPhone":"string //用户手机号码",
	"userEmail":"string //用户邮箱地址",
	"userHead":"string //用户头像图片地址",
	"userCreateTime":"string //用户创建时间"
}

返回结果

{
	"code":"int //错误码",
	"msg":"string //错误信息",
	"data":{}
}
